    <script>
        async function loadPrices() {
            const pricesElement = document.getElementById('prices');
            const lastUpdatedElement = document.getElementById('lastUpdated');
            
            pricesElement.innerHTML = '<p class="loading">Загрузка актуальных цен...</p>';
            
            try {
                const response = await fetch('http://localhost:5000/api/assets/with-prices');
                const assets = await response.json();
                
                let html = '';
                assets.forEach(asset => {
                    const change = asset.change || 0;
                    const changeColor = change >= 0 ? 'positive' : 'negative';
                    const changeSymbol = change >= 0 ? '📈' : '📉';
                    
                    html += `
                        <div class="price-item ${changeColor}">
                            <strong>${asset.name} (${asset.symbol})</strong><br>
                            <span style="font-size: 1.2em; font-weight: bold;">$${asset.price ? asset.price.toLocaleString('en-US', {minimumFractionDigits: 2, maximumFractionDigits: 6}) : 'N/A'}</span><br>
                            <span class="price-change" style="color: ${change >= 0 ? '#27ae60' : '#e74c3c'}">
                                ${changeSymbol} ${change ? change.toFixed(2) + '%' : 'N/A'} за 24ч
                            </span>
                        </div>
                    `;
                });
                
                pricesElement.innerHTML = html;
                lastUpdatedElement.textContent = `Последнее обновление: ${new Date().toLocaleTimeString()}`;
                
            } catch (error) {
                console.error('Ошибка:', error);
                pricesElement.innerHTML = `
                    <div style="text-align: center; color: #e74c3c; padding: 20px;">
                        <p>❌ Ошибка загрузки данных</p>
                        <p>Убедитесь, что API запущено на http://localhost:5000</p>
                        <p><small>Запустите API: <code>python app.py</code> в папке api</small></p>
                    </div>
                `;
            }
        }
        
        // Загружаем цены при открытии страницы
        loadPrices();
        
        // Автообновление каждые 30 секунд
        setInterval(loadPrices, 30000);
    </script>
